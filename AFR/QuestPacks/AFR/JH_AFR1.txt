Item _alch_ item class 17 subclass 11

Person _friend_ group Local_3.0 female

Place _tav_ local tavern
Place _res_ local house

Clock _timer_ 3.00:00

--	Quest start-up:
-- this will require a precursor quest with the location trigger
-    player currentmappixel x <xCoord> y <yCoord>
    say 1010
    log 1011 step 0
    start timer _timer_
    hide npc _dhunek_
    hide npc _friend_
    pc at _tav_ do _inside_

_inside_ task:
    prompt 1012 yes _begin_ no _quit_

_begin_ task:
    say 1013
    log 1014 step 0
    place npc _friend_ at _res_
    restore _friend_
    stop timer _timer_

_clickfriend_ task:
    clicked npc _friend_
    say 1015
    log 1016 step 1
    place item _alch_ at _dung_
    reveal _dung_

_getscales_ task:
    clicked item _alch_
    say 1017
    log 1018 step 1

_deliverscales_ task:
    toting _alch_ and _friend_ clicked
    say 1019
    log 1020 step 1

_reunion_ task:
    when _inside_ and _deliverscales_
    prompt 1021 yes _accompany_ no _quit2_

_convo_ task:
    when _accompany_ and not _inside_
    say 1023
    end quest

_quit_ task:
    say 1024
    end quest

_timer_ task:
  end quest

_accompany_ task:
    say 1025
    log 1022 step 0
    remove log step 1
    add _dhunek_ face

_quit2_ task:
    say 1026
    end quest
