    clear _noenter_

    when pc exits city
    clear _enter_

    start task _dospawn_

_dospawn_ task:
    say 1010
    create foe _enemy_ every 0 minutes 3 times with 100% success
    start timer _ending_

_ending_ task:
    end quest
