--   Quest start-up:
  start timer _timer_

_start_ task:
  clicked _qg_ and at least 200 gold otherwise do _fail_
  say 1010
  place item _sack_ at _res_
  pc at _res_ do _inside_
  log 1030 step 0

_fail_ task:
  say 1016

variable _inside_
_nosneak_ task:
  when not _inside_
  stop timer _sneak_

_entry_ task:
  when _inside_ and _start_ and not _day_
  say 1011
  start timer _sneak_

_day_ task:
  daily from 5:00 to 23:59

_caught1_ task:
  when _inside_ and _start_ and _day_
  say 1015
  setplayercrime Theft
  spawncityguards immediate

_sneak_ task:
  prompt 1012 yes _agree_ no _caught2_

_agree_ task:
  say 1013

_caught2_ task:
  say 1014
  setplayercrime Theft
  spawncityguards immediate

_found_ task:
  clicked item _sack_
  take _sack_ from pc
  give pc _reward_
  remove log step 0

_finish_ task:
  when _found_ and not _inside_
  end quest

_timer_ task:
  end quest
