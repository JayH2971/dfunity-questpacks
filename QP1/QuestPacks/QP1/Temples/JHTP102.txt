--	Quest start-up:
  start timer _timer_
  log 1030 step 0
  place npc _donor_ at _res_

_receive_ task:
  clicked npc _donor_
  say 1020
  get item _money_
  start timer _delay_
  _money_ used do _question_

_question_ task:
  prompt 1028 yes _thief_ no _dud_

_thief_ task:
  say 1029
  get item _steal_
  make _steal_ permanent
  take _money_ from pc
  change repute with _questgiver_ by -3
  end quest

_dud_ task:
  say 1031

_delay_ task:
  create foe _enemy_ every 300 minutes 3 times with 25% success

_choice_ task:
  injured _enemy_
  prompt 1025 yes _altend_ no _refuse_

_altend_ task:
  say 1027
  restrain foe _enemy_
  remove foe _enemy_
  take _money_ from pc
-  start quest ????
  end quest

  say 1026

_victory_ task:
  toting _money_ and _questgiver_ clicked
  take _money_ from pc
  give pc _reward_
  end quest

_timer_ task:
  say 1003
  end quest
