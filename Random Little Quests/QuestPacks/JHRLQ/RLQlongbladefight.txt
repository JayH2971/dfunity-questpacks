    say 1011
    end quest

    when skill LongBlade is at least 30

_pay_ task:
    pay 50 money do _accept_ otherwise do _nomoney_

_accept_ task:
    clear _pay_

_finalwin_ task:
    when _accept_ and _qualified_
    give pc _reward_
    end quest

_finalfail_ task:
    when _accept_ and not _qualified_
    say 1012
    reduce player health by 10
    end quest

_nomoney_ task:
    say 1013
    end quest
